js实现5种常见排序

2018-09-11

#简介：前端面试题中，通常会被问到javascript实现排序算法。博主之前都是用c/c++实现的排序，面试中突然被问到，有点措手不及，所以特地整理了一下常见的5种排序方法的js实现代码, 欢迎大家一起沟通交流。

#插入排序
function insertSort(arr){
var len = arr.length;
for(var i =1;i<len;i++){ //控制轮数
var key = arr[i]; //保留比较的数值
var index = i;
for(var j =i-1;j>=0;j–){ //往后挪位置，找到插入位置
if (arr[j]>key){
arr[j+1]=arr[j];
index = j;
}
}
arr[index] = key;
}
return arr;
}
var arr=[3,1,6,9];
console.log(insertSort(arr))

#选择排序
function selectSort(arr){
var len = arr.length;
for(var i =0;i<len;i++){
var index = i;
for(var j =i+1;j<len;j++){
if (arr[index]>arr[j]){
index = j;
}
}
var temp = arr[i];
arr[i]=arr[index];
arr[index]=temp;
}
return arr;
}
var arr=[1,9,6,3];
selectSort(arr);
console.log(arr)

#快速排序
function quickSort(arr, left, right) {
if (left > right) { //退出机制
return;
}
let i = left;
let j = right;
let temp = arr[left];
while (i < j) {
while (arr[j] > temp && i < j) {
j–;
}
while (arr[i] <= temp && i < j) { //关键点=不能少
i++;
}
if (i < j) {
let t = arr[i];
arr[i] = arr[j];
arr[j] = t;
}
}
arr[left] = arr[i];
arr[i] = temp;
//自调用
this.quickSort(arr, left, i - 1);
this.quickSort(arr, i + 1, right);
}
var arr = [3, 9, 1, 6];
quickSort(arr,0,3)
console.log(arr);

#冒泡排序
function bubbleSort(arr){
var len = arr.length;
for(var i =0;i<len-1;i++){ //控制轮数
for(var j =i;j<len;j++){
if (arr[j]>arr[j+1]){
var temp = arr[j];
arr[j] = arr[j+1];
arr[j+1] = temp;
}
}
}
return arr;
}
var arr =[3,1,6,2];
bubbleSort(arr)
console.log(arr)

#堆排序
//调整最大堆
function adjust(arr, length, idx) { //数组下标从0开始
let maxIdx = idx;
let left = idx 2 + 1;
let right = idx 2 + 2
if (left >= length && right >= length) { //退出机制
return;
}
if (arr[maxIdx] < arr[left] && left < length) {
maxIdx = left;
}
if (arr[maxIdx] < arr[right] && right < length) {
maxIdx = right;
}
if (maxIdx != idx) {
this.swap(arr, maxIdx, idx)
this.adjust(arr, length, maxIdx)
}
}

function swap(arr, a, b) { //数组的交换，不能直接a,b进行交换
let tmp = arr[a];
arr[a] = arr[b];
arr[b] = tmp;
}

//堆排序
function heapSort(arr, len) {
for (let i = parseInt(len / 2) - 1; i >= 0; i–) {
this.adjust(arr, len, i)
}
for (let j = len - 1; j >= 1; j–) {
this.swap(arr, j, 0); //都跟arr[0]进行互换
this.adjust(arr, j, 0) //注意 j，调整的长度不包括已经排好序的末尾数组
}
}
var arr = [9, 2, 1, 3];
heapSort(arr, 4)
console.log(arr)

docker相关知识

2018-08-09

#简介

  1. 什么是docker？
    NodeJs解释：将应用程序及其依赖以一种标准的单位进行打包。换言之，docker是一个快速部署的轻量级虚拟技术项目，开发人员可以一起打包程序和运行环境，成为一个镜像，只需要下载镜像文件就可以将程序跑起来，免去每次都安装各种依赖和环境的麻烦，还能够做到应用程序之间的隔离。粗糙地认为类似于虚拟机，但不同于虚拟机，

  2. 涉及的概念：镜像，容器，registry，

  3. 必备知识：linux命令行，bash

  4. 安装和配置： 去官网（http://www.docker.com/）下载对应的docker软件（本人使用windows版本）。 右键docker选择settings–》Docker Deamon–》registry-mirrors 添加加速器连接 http://466ed505.m.daocloud.io/ （来自于阿里云或者网易云的docker加速器，提高docker下载速度）

  5. 本地运行: 打开命令行窗口，进行docker相关操作（最好跟着下面的指令在自己的电脑上敲一遍，看一下效果）

    5.1 基本操作：查看，启动和停止容器，下载镜像文件
    5.1.1 docker images //查看镜像列表
    5.1.2 docker ps //查看当前运行中的容器信息
    5.1.3 docker ps -a //查看所有容器信息，包括运行中和暂停的容器信息
    5.1.4 docker pull nginx //下载nginx镜像文件
    5.1.5 docker run -itd -p 10090:80 –name “my-nginx” nginx 

    //创建并启动nginx的新容器，并把虚拟机的80端口映射到本机的10090端口，在浏览器中可通过localhost:10090进行访问到nginx

    5.1.6 docker stop #containerId //停止指定的容器
    5.1.7 docker start #containerId //重启指定的容器，通过docker ps/docker ps -a 查看容器的

    5.2 在运行的容器内执行bash命令
    5.2.1 docker exec -it #containerId bash //在运行的容器中执行bash命令
    5.2.2 cat /etc/nginx/conf.d/default.conf //可以执行下述命令，如查看nginx镜像文件的配置文件
    5.2.3 apt-get update //
    5.2.4 apt-get install vim //安装vim
    5.2.5 vim /usr/share/nginx/html/index.html 打开步骤10看到的html文件，进行修改
    5.2.6 exit 退出

    5.3 提交修改后生成的新的镜像文件
    5.3.1 docker ps //查看
    5.3.2 docker diff
    5.3.3 docker commit -m “add: 修改了nginx首页的效果” containerId nginx2:1.0.1 

    //imageName自定义，tag也是 ,docker容器内的修改是暂存的，相当于新的imgage,需要进行保存

    5.3.4 docker images //查看镜像列表，新添加了一个镜像nginx2

    5.4 删除容器操作
    5.4.1 dokcer stop containerId //停止指定容器
    5.4.2 docker ps -a //查看运行中的容器
    5.4.3 docker rm containerId //删除容器
    5.4.4 删除镜像：docker rmi imageId 前提必须先停止该镜像所在容器，然后删除容器，再删除镜像，若镜像被多个容器引用，需要强制删除,指令后面再 –force

    5.5 利用docker部署NodeJs应用，非常轻量和方便（还有问题，待补充）
    5.5.1 docker search node
    5.5.2 docker pull node //下载node镜像文件
    5.5.3 //用nodejs编写一个简单的hello world后端文件app.js，docker images查看到node是否下载成功
    5.5.4 docker run -itd -p 10092:82 –name “my-node” node //为node镜像文件创建新的容器 //打开浏览器localhost:10092可以查看hello world效果
    5.5.5 docker cp app.js containerId:/home/hello //将该文件拷贝到制定的容器内，并指定路径

总结：使用docker的好处之一，就是方便学习新东西，用docker pull下载镜像文件，并在独立的容器内进行操作，用完后删掉容器和相应的文件，不影响本地
